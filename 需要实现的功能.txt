2. 必备权限文件（entitlements.plist 固定内容）
xml
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
    <!-- 关闭沙盒 核心必备 -->
    <key>com.apple.private.security.no-sandbox</key>
    <true/>
    <!-- 全屏屏幕截图权限 -->
    <key>com.apple.private.screencapture.capture-all-displays</key>
    <true/>
    <!-- 辅助功能模拟点击核心权限 -->
    <key>com.apple.private.accessibility.server</key>
    <true/>
    <!-- 支持Lua动态代码运行 -->
    <key>dynamic-codesigning</key>
    <true/>
    <!-- 跳过库校验，巨魔运行更稳定 -->
    <key>com.apple.private.skip-library-validation</key>
    <true/>
</dict>
</plist>
3. Lua 环境集成框架
编译产物：liblua.a arm64 iOS 静态库
导入头文件：lua.h lualib.h lauxlib.h
基础生命周期
初始化虚拟机：luaL_newstate()
加载基础库：luaL_openlibs()
注册自定义原生接口
执行脚本：luaL_dostring()
销毁虚拟机释放内存
运行限制：单进程单 Lua 虚拟机，禁止多脚本并发冲突
4. 所有对外开放 Lua 接口框架（固定命名，统一规范）
4.1 屏幕图像类接口
lua
-- 1. 获取指定坐标RGB颜色
r, g, b = get_screen_color(x, y)

-- 2. 区域模糊找色，返回第一个匹配坐标，无匹配返回0,0
find_x, find_y = find_color(x1, y1, x2, y2, target_r, target_g, target_b, similarity)
-- 参数说明：
-- x1y1起始坐标 x2y2结束坐标 目标RGB 相似度0-100

5. 原生开发核心功能规则
截图功能：使用CGDisplayStream实现，后台静默截图，不弹窗
找色逻辑：纯内存像素遍历，不保存图片文件，运行高效不卡顿
点击逻辑：基于 iOS 原生辅助功能 API，无注入、无 Hook，稳定兼容全机型
分辨率：自动获取手机真实屏幕分辨率，脚本坐标自动适配
脚本安全：加入死循环检测、语法错误捕获，APP 不会闪退
6. 标准测试 Lua 脚本模板（直接可用）
lua
-- iOS巨魔自动化循环找色点击脚本
while true do
    -- 在屏幕200,300 至 600,900 范围找纯白色
    local tx, ty = find_color(200, 300, 600, 900, 255, 255, 255, 90)
    if tx > 0 and ty > 0 then
        log("成功找到目标坐标："..tx..","..ty)
        click(tx, ty)
        sleep(800)
    end
    -- 循环间隔
    sleep(300)
end
